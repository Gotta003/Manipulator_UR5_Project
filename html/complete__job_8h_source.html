36class GripperCommunicator : public rclcpp::Node {
37public:
38 GripperCommunicator() : Node("gripper_communicator") {
39 open_gripper_client_ = this->create_client<std_srvs::srv::Trigger>("open_gripper");
40 close_gripper_client_ = this->create_client<std_srvs::srv::Trigger>("close_gripper");
41 rotation_gripper_client_=this->create_client<ros2_ur5_interface::srv::SetFloat64>("rotate_gripper");
42 while(!open_gripper_client_->wait_for_service(std::chrono::seconds(5))) {
43 RCLCPP_ERROR(this->get_logger(), "Service open_gripper not available.");
44 }
45 while(!close_gripper_client_->wait_for_service(std::chrono::seconds(5))) {
46 RCLCPP_ERROR(this->get_logger(), "Service close_gripper not available.");
47 }
48 while(!rotation_gripper_client_->wait_for_service(std::chrono::seconds(5))) {
49 RCLCPP_ERROR(this->get_logger(), "Service rotate_gripper not available.");
50 }
51
52 }
53 void open() {
54 callService(open_gripper_client_, "Opening the gripper");
55 }
56
57 void close() {
58 callService(close_gripper_client_, "Closing the gripper");
59 }
60 void rotate_gripper(float rot) {
61 callServiceRotate(rotation_gripper_client_, rot, "Rotating Service Sending");
62 }
63private:
64 rclcpp::Client<std_srvs::srv::Trigger>::SharedPtr open_gripper_client_;
65 rclcpp::Client<std_srvs::srv::Trigger>::SharedPtr close_gripper_client_;
66 rclcpp::Client<ros2_ur5_interface::srv::SetFloat64>::SharedPtr rotation_gripper_client_;
67 template<typename ServiceClient> void callServiceRotate(ServiceClient client, float rot, const std::string &action) {
68 RCLCPP_INFO(this->get_logger(), "%s", action.c_str());
69 auto request=std::make_shared<ros2_ur5_interface::srv::SetFloat64::Request>();
70 request->data=rot;
71 auto future=client->async_send_request(request);
72 if(rclcpp::spin_until_future_complete(this->get_node_base_interface(), future)==rclcpp::FutureReturnCode::SUCCESS) {
73 auto response=future.get();
74 if(response->success) {
75 RCLCPP_INFO(this->get_logger(), "%s: Rotation Sent Successfully", action.c_str());
76 }
77 else {
78 RCLCPP_WARN(this->get_logger(), "%s: Rotation Went Wrong", action.c_str());
79 }
80 }
81 else {
82 RCLCPP_ERROR(this->get_logger(), "%s service call timed out.", action.c_str());
83 }
84 }

86 template<typename ServiceClient>
87 void callService(ServiceClient client, const std::string &action) {
88 RCLCPP_INFO(this->get_logger(), "%s", action.c_str());
89 auto request = std::make_shared<std_srvs::srv::Trigger::Request>();
90 auto future = client->async_send_request(request);
91 if (rclcpp::spin_until_future_complete(this->get_node_base_interface(), future) ==
92 rclcpp::FutureReturnCode::SUCCESS) {
93 auto response = future.get();
94 if (response->success) {
95 RCLCPP_INFO(this->get_logger(), "%s succeeded: %s", action.c_str(), response->message.c_str());
96 } else {
97 RCLCPP_WARN(this->get_logger(), "%s failed: %s", action.c_str(), response->message.c_str());
98 }
99 } else {
100 RCLCPP_ERROR(this->get_logger(), "%s service call timed out.", action.c_str());
101 }
102 }
